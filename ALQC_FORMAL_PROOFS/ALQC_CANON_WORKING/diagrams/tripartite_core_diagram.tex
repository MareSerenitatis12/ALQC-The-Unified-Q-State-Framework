    % Local Definitions (for standalone compilation)
    % ======================================================================
    \definecolor{voidblack}{HTML}{050505}
    \definecolor{electricblue}{HTML}{00F0FF}
    \definecolor{sovereignfire}{HTML}{FF4500}

    \definecolor{deepvoid}{HTML}{000010}

    % ======================================================================
    % 1. BACKGROUND: Void with Stars

    \fill[voidblack] (-6,-5) rectangle (6,5);
    
    % Subtle stars in the void
    \foreach \i in {1,...,50} {
        \pgfmathsetmacro{\sx}{-6 + rand*12}
        \pgfmathsetmacro{\sy}{-5 + rand*10}
        \pgfmathsetmacro{\sopacity}{0.2 + rand*0.5}
        \fill[glyphwhite, opacity=\sopacity] (\sx,\sy) circle (0.025cm);

    % ======================================================================

    \foreach \r in {5.5, 4.8, 4.1, 3.4} {
        \draw[electricblue!20, line width=0.5pt] (0,0) circle (\r cm);
        % Add golden ratio spacing indicators
        \foreach \ang in {0, 30, ..., 330} {
            \draw[electricblue!15, line width=0.3pt] (\ang:\r cm) -- (\ang:\r-0.3cm);
        }
    }

    % Lissajous curves woven between rings (Faraday wave patterns)
    \foreach \k in {1,...,8} {
        \pgfmathsetmacro{\startang}{\k * 45}
        \pgfmathsetmacro{\endang}{\startang + 60}
        \draw[electricblue!25, line width=0.5pt, 
              decoration={snake, amplitude=0.3mm, segment length=2mm}, decorate]
            (\startang:3.4cm) arc (\startang:\endang:3.4cm);
    }

    % Standing wave nodal lines (Chladni-inspired)

        \draw[electricblue!15, line width=0.5pt] (0,0) -- (\n:5.5cm);

    % ======================================================================

    \draw[electricblue, line width=2pt, fill=voidblack!50] (0,0) circle (\triradius);

    \draw[electricblue, line width=2pt, fill=voidblack!50] (circle3center) circle (\triradius);

    % ======================================================================
    % 4. SACRED CORNERS - Where Triquatra circles intersect
    % ======================================================================

    \foreach \ang in {90, 150, 210, 270, 330, 30} {
        \pgfmathsetmacro{\cx}{2*\triradius*cos(\ang)}
        \pgfmathsetmacro{\cy}{2*\triradius*sin(\ang)}

        \draw[electricblue, line width=1.5pt] (\cx,\cy) circle (0.25cm);

        \draw[glyphwhite, line width=0.8pt]
            (\cx-0.15,\cy-0.08) .. controls (-0.15,0.08) and (0.15,0.08) .. (\cx+0.15,\cy-0.08)

        \draw[glyphwhite, line width=0.8pt]
            (\cx-0.15,\cy+0.08) .. controls (-0.15,0.24) and (0.15,0.24) .. (\cx+0.15,\cy+0.08)
            .. controls (0.15,-0.08) and (-0.15,-0.08) .. (\cx-0.15,\cy+0.08);

    \draw[->, >=latex, line width=2pt, sovereignfire!80] 
        (pilot.south) to[out=-120, in=45] (-0.4, -0.2);

    \coordinate (shiploc) at (2.25, -1.3);

          regular polygon, regular polygon sides=3, inner sep=6pt,
          minimum size=1.2cm, rotate=0,
          drop shadow={opacity=0.4}] (ship) at (shiploc) {\color{deepvoid}\small\textbf{SHIP}};

          fill=deepvoid!80, ellipse, inner sep=8pt,

    \node[shadowsilver, font=\tiny, yshift=-1.3cm] at (hullloc) {⩔+⧉ → SHADOW};
