Architecture:

Usage:

    summary = generate_summary(chunk_text)
    payload = f"[{summary}] {chunk_text}"
"""

# Model node configuration - supports llama.cpp and ollama backends
# Set via environment variable SEMANTIC_SUMMARY_NODE_JSON or override in code
# Example for llama.cpp:
#   MODEL_NODE = {"url": "http://127.0.0.1:11444", "model": "default", "type": "llama", "workers": 4}
# Example for ollama:
#   MODEL_NODE = {"url": "http://127.0.0.1:11434", "model": "llama3.2:latest", "type": "ollama", "workers": 1}
#
# Workers: Number of concurrent requests to the LLM backend
#   - llama.cpp: Can handle multiple concurrent requests (default: 4)
#   - ollama: Single-threaded by default, set workers=1 unless using OLLAMA_NUM_PARALLEL (default: 1)

# Support multiple nodes - can be a single dict or list of dicts
# Set via environment variable SEMANTIC_SUMMARY_NODES_JSON or override in code
# Example for multiple ollama instances:
#   MODEL_NODES = [
#       {"url": "http://127.0.0.1:11444", "model": "lfm2.5-thinking:latest", "type": "ollama", "workers": 2},
#       {"url": "http://127.0.0.1:11445", "model": "lfm2.5-thinking:latest", "type": "ollama", "workers": 4}
#   ]
_default_nodes_json = os.environ.get(
    "SEMANTIC_SUMMARY_NODES_JSON",
    # NOTE: sam860/lfm2.5:1.2b-Q8_0 does NOT exist on either instance - use lfm2.5-thinking:latest
    '[{"url": "http://127.0.0.1:11444", "model": "ministral-3:3b-cloud", "type": "ollama", "workers": 1}, '
    '{"url": "http://127.0.0.1:11445", "model": "ministral-3:3b-cloud", "type": "ollama", "workers": 4}]'
)
MODEL_NODES = json.loads(_default_nodes_json)
# Ensure it's always a list
if isinstance(MODEL_NODES, dict):
    MODEL_NODES = [MODEL_NODES]

# Round-robin node selection
_current_node_index = 0

def get_next_node() -> dict:
    """Get next node in round-robin fashion."""
    global _current_node_index
    node = MODEL_NODES[_current_node_index % len(MODEL_NODES)]
    _current_node_index += 1
    return node

# Concurrency semaphore - sum of all workers across nodes
_max_workers = sum(node.get("workers", 1) for node in MODEL_NODES)
_WORKER_SEMAPHORE = asyncio.Semaphore(_max_workers)

# Summary generation settings
SUMMARY_MAX_TOKENS = 50  # Keep summaries short
SUMMARY_TIMEOUT = 5  # Seconds to wait for response
SUMMARY_PROMPT = """Summarize the following text in exactly one concise sentence. Do not add any preamble or explanation. Just the summary:

{text}

Summary:"""


def _build_llama_payload(prompt: str, node: dict) -> dict:
    """Build payload for llama.cpp backend with ALQC context as system prompt."""
    return {
        "model": node.get("model", "default"),
        "prompt": prompt,
        "system_prompt": ALQC_CONTEXT.strip(),  # Inject ALQC context as system prompt
        "n_predict": SUMMARY_MAX_TOKENS,
        "temperature": 0.3,
        "stop": ["\n", "\n\n", "Summary:", "Text:"]
    }


def _build_ollama_payload(prompt: str, node: dict) -> dict:
    """Build payload for ollama backend with ALQC context as system prompt."""
    return {
        "model": node.get("model", "llama3.2:latest"),
        "prompt": prompt,
        "system": ALQC_CONTEXT.strip(),  # Inject ALQC context as system prompt
        "stream": False,
        "options": {
            "num_predict": SUMMARY_MAX_TOKENS,
            "temperature": 0.3,
            "stop": ["\n", "\n\n", "Summary:", "Text:"]
        }
    }


def _get_endpoint(node: dict) -> str:
    """Get the API endpoint based on backend type."""
    backend_type = node.get("type", "llama")
    base_url = node.get("url", "http://127.0.0.1:11444")
    
    if backend_type == "ollama":
        return f"{base_url}/api/generate"
    else:  # llama.cpp
        return f"{base_url}/completion"


def _extract_content(result: dict, node: dict) -> str:
    """Extract content from response based on backend type."""
    backend_type = node.get("type", "llama")
    
    if backend_type == "ollama":
        return result.get("response", "").strip()
    else:  # llama.cpp
        return result.get("content", "").strip()


async def generate_summary_async(text: str, max_chars: int = 100) -> str:
    """
    Generate a 1-sentence semantic summary using configured LLM backend.
    
    Args:
        text: The chunk text to summarize
        max_chars: Maximum characters for the summary (default 100)
    
    Returns:
        A 1-sentence summary, or empty string on failure
    """
    if not text or not text.strip():
        return ""
    
    # Use the chunk text as-is (main script handles chunking)
    input_text = text.strip()
    
    # Get next node in round-robin fashion
    node = get_next_node()
    prompt = SUMMARY_PROMPT.format(text=input_text)
    backend_type = node.get("type", "llama")
    
    # Build payload based on backend type
    if backend_type == "ollama":
        payload = _build_ollama_payload(prompt, node)
    else:
        payload = _build_llama_payload(prompt, node)
    
    endpoint = _get_endpoint(node)
    
    # Use semaphore to limit concurrent requests based on workers config
    async with _WORKER_SEMAPHORE:
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    endpoint,
                    json=payload,
                    timeout=aiohttp.ClientTimeout(total=SUMMARY_TIMEOUT)
                ) as resp:
                    if resp.status != 200:
                        return ""
                    
                    result = await resp.json()
                    summary = _extract_content(result, node)
                    
                    # Strip thinking tokens from thinking models (e.g., lfm2.5-thinking)
                    # Removes <think>...</think> blocks
                    import re
                    summary = re.sub(r'<think>.*?</think>', '', summary, flags=re.DOTALL)
                    summary = summary.strip()
                    
                    # Clean up the summary
                    summary = summary.strip('"\'')
                    
                    # Truncate if too long
                    if len(summary) > max_chars:
                        # Try to cut at last complete sentence
                        last_period = summary.rfind('.', 0, max_chars)
                        if last_period > max_chars // 2:
                            summary = summary[:last_period + 1]
                        else:
                            summary = summary[:max_chars - 3] + "..."
                    
                    return summary
                    
        except asyncio.TimeoutError:
            return ""
        except Exception as e:
            # Silent fail - return empty string
            return ""


def generate_summary(text: str, max_chars: int = 100) -> str:
    """
    Synchronous wrapper for generate_summary_async.
    
    Args:
        text: The chunk text to summarize
        max_chars: Maximum characters for the summary
    
    Returns:
        A 1-sentence summary, or empty string on failure
    """
    try:
        loop = asyncio.get_event_loop()
        if loop.is_running():
            # We're in an async context, create a new loop
            import concurrent.futures
            with concurrent.futures.ThreadPoolExecutor() as executor:
                future = executor.submit(
                    asyncio.run,
                    generate_summary_async(text, max_chars)
                )
                return future.result(timeout=SUMMARY_TIMEOUT + 1)
        else:
            return loop.run_until_complete(generate_summary_async(text, max_chars))
    except Exception:
        return ""


async def generate_summary_batch_async(texts: list[str], max_chars: int = 100) -> list[str]:
    """
    Generate summaries for multiple texts concurrently.
    
    The main script handles batching - this just processes what it receives.
    The _WORKER_SEMAPHORE limits concurrent LLM requests.
    
    Args:
        texts: List of chunk texts to summarize (already batched by main script)
        max_chars: Maximum characters per summary
    
    Returns:
        List of summaries (same order as input)
    """
    if not texts:
        return []
    
    # Semaphore in generate_summary_async limits concurrent LLM requests
    tasks = [generate_summary_async(text, max_chars) for text in texts]
    return await asyncio.gather(*tasks)


# ==========================================
# FALLBACK: Fast first-sentence extraction
# ==========================================
def extract_first_sentence(text: str, max_chars: int = 100) -> str:
    """
    Fast fallback: Extract first sentence from text.
    No LLM call - used when SEMANTIC_SUMMARY_ENABLE is False or LLM fails.
    
    Args:
        text: The chunk text
        max_chars: Maximum characters
    
    Returns:
        First sentence or truncated text
    """
    import re
    
    if not text or not text.strip():
        return ""
    
    text = text.strip()
    
    # Find first sentence (ends with . ! ? followed by space or end)
    match = re.search(r'^.*?[.!?](?:\s|$)', text)
    if match:
        sentence = match.group(0).strip()
        if len(sentence) <= max_chars:
            return sentence
        return sentence[:max_chars - 3] + "..."
    
    # No sentence ending - take first max_chars
    if len(text) <= max_chars:
        return text
    return text[:max_chars - 3] + "..."


# ==========================================
# MAIN ENTRY POINT
# ==========================================
def get_semantic_summary(text: str, max_chars: int = 100, use_llm: bool = True) -> str:
    """
    Main entry point for semantic summary generation.
    
    Args:
        text: The chunk text to summarize
        max_chars: Maximum characters for the summary
        use_llm: If True, use LLM backend; if False, use fast extraction
    
    Returns:
        A 1-sentence summary
    """
    if use_llm:
        summary = generate_summary(text, max_chars)
        if summary:
            return summary
        # Fallback to extraction if LLM fails
        return extract_first_sentence(text, max_chars)
    return extract_first_sentence(text, max_chars)
